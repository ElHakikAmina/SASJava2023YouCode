
 CREATE DATABASE Bibliotheque;

USE Bibliotheque;

CREATE TABLE livre (
    ISBN VARCHAR(13) PRIMARY KEY,
    titre VARCHAR(255) NOT NULL,
    auteur VARCHAR(255),
   quantite int
);


CREATE TABLE Emprunteur (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nom_complet VARCHAR(255) NOT NULL,
    numTel int,
    adresse VARCHAR(255)
);


CREATE TABLE emprunt_livre (
    id INT PRIMARY KEY AUTO_INCREMENT,
    livreISBN VARCHAR(13),
    emprunteurID INT,
    dateEmprunt DATE,
    dateRetour DATE,
    FOREIGN KEY (LivreISBN) REFERENCES livre(ISBN),
    FOREIGN KEY (EmprunteurID) REFERENCES emprunteur(ID)
);
